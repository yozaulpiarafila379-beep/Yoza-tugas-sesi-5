🧬 4 Pilar OOP

1. Enkapsulasi

Stok dan kondisi barang hanya diubah lewat metode tambahStok() dan kurangiStok().

2. Pewarisan (Inheritance)

Barang mewarisi Entitas. TransaksiMasuk & TransaksiKeluar mewarisi Transaksi.

3. Polimorfisme

tampilkanInfo() dan prosesTransaksi() di-override pada turunan.

4. Interaksi Objek

Objek Barang dan Pelanggan dikelola oleh Gudang.

abstract class Entitas {
    String kode, nama;
    Entitas(this.kode, this.nama);
    void tampilkanInfo();
    void cetakIdentitas() {
        print("🔹 $kode — $nama");
    }
}

class Barang extends Entitas {
    String kategori, merk, satuan, kondisi;
    int stok;
    double hargaBeli, hargaJual;

    Barang(String kode, String nama, this.kategori, this.merk,
        this.stok, this.satuan, this.kondisi,
        this.hargaBeli, this.hargaJual) : super(kode, nama);

    @override void tampilkanInfo() {
        print("📦 $kode | $nama | Stok: $stok $satuan");
    }

    void tambahStok(int j) { stok += j; print("✅ +$j $satuan. Total: $stok"); }

    void kurangiStok(int j) {
        if (stok >= j) { stok -= j; print("✅ -$j $satuan. Total: $stok"); }
        else print("❌ Stok tidak cukup! Tersisa: $stok $satuan");
    }
}

class Pelanggan extends Entitas {
    String alamat, telepon, email;
    Pelanggan(String kode, String nama, this.alamat, this.telepon, this.email) : super(kode, nama);
    @override void tampilkanInfo() {
        print("👤 $kode | $nama | $alamat | $telepon");
    }
}

class Gudang {
    List<Barang> daftarBarang = [];
    List<Pelanggan> daftarPelanggan = [];

    void tambahBarang(Barang b) { daftarBarang.add(b); }
    void tambahPelanggan(Pelanggan p) { daftarPelanggan.add(p); }

    Barang? cariBarang(String k) {
        for (var b in daftarBarang) if (b.kode == k) return b;
        return null;
    }

    Pelanggan? cariPelanggan(String k) {
        for (var p in daftarPelanggan) if (p.kode == k) return p;
        return null;
    }

    void transaksiMasuk(String k, int j) {
        var b = cariBarang(k);
        if (b == null) { print("❌ Barang tidak ditemukan!"); return; }
        b.tambahStok(j);
    }

    void transaksiKeluar(String k, int j) {
        var b = cariBarang(k);
        if (b == null) { print("❌ Barang tidak ditemukan!"); return; }
        b.kurangiStok(j);
    }

    void beli(String idPelanggan, String kodeBarang, int j) {
        var b = cariBarang(kodeBarang);
        var p = cariPelanggan(idPelanggan);
        if (b == null || p == null) { print("❌ Data tidak ditemukan!"); return; }
        if (b.stok < j) { print("❌ Stok tidak cukup!"); return; }
        b.stok -= j;
        print("✅ ${p.nama} membeli ${b.nama} $j ${b.satuan}");
    }

    void tampilkanSemuaBarang() {
        print("\n📦 DAFTAR BARANG");
        for (var b in daftarBarang) b.tampilkanInfo();
        print("Total: ${daftarBarang.length}\n");
    }

    void tampilkanSemuaPelanggan() {
        print("\n👤 DAFTAR PELANGGAN");
        for (var p in daftarPelanggan) p.tampilkanInfo();
        print("Total: ${daftarPelanggan.length}\n");
    }

    void menu() {
        String? buffer;
        int idx = 0;
        String input = '''
        1
        2
        3
        BRG001
        10
        4
        BRG003
        5
        5
        PLG002
        BRG002
        3
        6
        0
        ''';

        void setInput(String s) { buffer = s; idx = 0; }
        String? readLine() {
            if (buffer == null || idx >= buffer!.length) return null;
            int start = idx;
            while (idx < buffer!.length && buffer![idx] != '\n') idx++;
            String result = buffer!.substring(start, idx);
            if (idx < buffer!.length && buffer![idx] == '\n') idx++;
            return result;
        }

        setInput(input);
        bool run = true;

        while (run) {
            print('\n===== MENU =====');
            print('1.Barang 2.Pelanggan 3.Masuk 4.Keluar 5.Beli 6.Stok 0.Keluar');
            print('Pilih: ');

            String? p = readLine();
            if (p == null) break;

            switch (p.trim()) {
                case '1': tampilkanSemuaBarang(); break;
                case '2': tampilkanSemuaPelanggan(); break;
                case '3':
                    print('Kode: '); String k = readLine() ?? '';
                    print('Jumlah: '); int j = int.tryParse(readLine() ?? '0') ?? 0;
                    transaksiMasuk(k, j); break;
                case '4':
                    print('Kode: '); String k = readLine() ?? '';
                    print('Jumlah: '); int j = int.tryParse(readLine() ?? '0') ?? 0;
                    transaksiKeluar(k, j); break;
                case '5':
                    print('ID Pelanggan: '); String id = readLine() ?? '';
                    print('Kode Barang: '); String k = readLine() ?? '';
                    print('Jumlah: '); int j = int.tryParse(readLine() ?? '0') ?? 0;
                    beli(id, k, j); break;
                case '6': laporanStok(); break;
                case '0':
                    run = false;
                    print('👋 Selesai!');
                    break;
                default: print('❌ Salah!');
            }
        }
    }

    void laporanStok() {
        print('\n📦 LAPORAN STOK');
        double total = 0;
        for (var b in daftarBarang) {
            double nilai = b.stok * b.hargaBeli;
            total += nilai;
            print("${b.kode} | ${b.nama} | ${b.stok} ${b.satuan} | Rp${nilai.toStringAsFixed(0)}");
        }
        print("Total Nilai Stok: Rp${total.toStringAsFixed(0)}\n");
    }
}

void main() {

}
