🧬 4 Pilar OOP

1. Enkapsulasi

Stok dan kondisi barang hanya diubah lewat metode tambahStok() dan kurangiStok().

2. Pewarisan (Inheritance)

Barang extends EntitasGudang. BarangMasuk & BarangKeluar extends Transaksi.

3. Polimorfisme

tampilkanInfo() dan prosesTransaksi() di-override pada turunan.

4. Interaksi Objek

Objek Barang berinteraksi dengan StokBarang dan Transaksi.

/// Class EntitasGudang - Superclass
class EntitasGudang {
  String kode;
  String nama;

  EntitasGudang({
    required this.kode,
    required this.nama,
  });

  void tampilkanInfo() {
    print('Kode: $kode');
    print('Nama: $nama');
  }

  void cetakIdentitas() {
    print('Entitas Gudang: $nama ($kode)');
  }
}

/// Class Barang extends EntitasGudang
class Barang extends EntitasGudang {
  String kategori;
  String merk;
  int jumlahStok;
  String satuan;
  String kondisi;

  Barang({
    required String kode,
    required String nama,
    required this.kategori,
    required this.merk,
    required this.jumlahStok,
    required this.satuan,
    required this.kondisi,
  }) : super(kode: kode, nama: nama);

  void tambahStok(int jumlah) {
    if (jumlah > 0) {
      this.jumlahStok += jumlah;
      print('✅ Stok bertambah: +$jumlah $satuan');
    }
  }

  bool kurangiStok(int jumlah) {
    if (this.jumlahStok >= jumlah && jumlah > 0) {
      this.jumlahStok -= jumlah;
      print('✅ Stok berkurang: -$jumlah $satuan');
      return true;
    } else {
      print('❌ Stok tidak mencukupi!');
      return false;
    }
  }

  @override
  void tampilkanInfo() {
    super.tampilkanInfo();
    print('Kategori: $kategori');
    print('Merk: $merk');
    print('Jumlah Stok: $jumlahStok $satuan');
    print('Kondisi: $kondisi');
  }
}

/// Class Transaksi
class Transaksi {
  String idTransaksi;
  String tanggal;
  String kodeBarang;
  int jumlah;

  Transaksi({
    required this.idTransaksi,
    required this.tanggal,
    required this.kodeBarang,
    required this.jumlah,
  });

  void prosesTransaksi() {
    print('🔄 Memproses transaksi $idTransaksi...');
  }

  void cetakHeader() {
    print('=' * 40);
    print('TRANSAKSI: $idTransaksi');
    print('Tanggal: $tanggal');
    print('Kode Barang: $kodeBarang');
    print('=' * 40);
  }
}

/// Class BarangMasuk extends Transaksi
class BarangMasuk extends Transaksi {
  String namaSupplier;

  BarangMasuk({
    required String idTransaksi,
    required String tanggal,
    required String kodeBarang,
    required int jumlah,
    required this.namaSupplier,
  }) : super(
          idTransaksi: idTransaksi,
          tanggal: tanggal,
          kodeBarang: kodeBarang,
          jumlah: jumlah,
        );

  @override
  void prosesTransaksi() {
    super.prosesTransaksi();
    print('📥 Barang Masuk dari: $namaSupplier');
    print('Jumlah: $jumlah unit');
  }
}

/// Class BarangKeluar extends Transaksi

/// Class Pengguna
class Pengguna {
  String username;
  String password;

  Pengguna({
    required this.username,
    required this.password,
  });

  bool login(String inputUsername, String inputPassword) {
    return username == inputUsername && password == inputPassword;
  }

  void tampilkanIdentitas() {
    print('👤 Pengguna: $username');
  }
}

/// Class User extends Pengguna
class User extends Pengguna {
  String level;

  User({
    required String username,
    required String password,
    required this.level,
  }) : super(username: username, password: password);

  @override
  bool login(String inputUsername, String inputPassword) {
    bool result = super.login(inputUsername, inputPassword);
    if (result) {
      print('✅ Login sebagai User Level: $level');
    }
    return result;
  }
}

/// Class StokBarang
class StokBarang {
  int stokAwal;
  int barangMasuk;
  int barangKeluar;
  int stokAkhir;
  String status;

  StokBarang({
    required this.stokAwal,
    this.barangMasuk = 0,
    this.barangKeluar = 0,
    this.stokAkhir = 0,
    this.status = 'Normal',
  });

  void hitungStok() {
    stokAkhir = stokAwal + barangMasuk - barangKeluar;
    if (stokAkhir < 10 && stokAkhir > 0) {
      status = '⚠️ Stok Menipis';
    } else if (stokAkhir == 0) {
      status = '❌ Stok Habis';
    } else {
      status = '✅ Stok Normal';
    }
  }

  void tampilkanLaporan() {
    print('\n📊 LAPORAN STOK');
    print('=' * 40);
    print('Stok Awal   : $stokAwal');
    print('Barang Masuk: +$barangMasuk');
    print('Barang Keluar: -$barangKeluar');
    print('-' * 40);
    print('Stok Akhir  : $stokAkhir');
    print('Status      : $status');
    print('=' * 40);
  }
